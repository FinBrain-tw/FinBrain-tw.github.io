網頁的差異

index 靜態網頁 動態網頁
原理 純粹使用HTML、CSS做出的單獨網頁檔案 透過javaScript非同步的方式去跟伺服器資料庫索取資料
優點 技術成分簡單，載入速度快 可因應不同需求展現內容、頁面管理較有效率不需要reload
缺點 無法因應不同需求展現不同內容 技術上較為複雜，載入速度中等
範例 cartlidgesqualitymeats.com Hahow
網頁原始碼 有關鍵字 無關鍵字
開啟Javascript Switcher 正常顯示 部分空白
爬蟲難度 容易 困難

每日股價爬蟲

  • 檢查網頁型態 ➜ 表格 ➜ pd.read

import requests

#json改成csv
url = 'https://www.twse.com.tw/exchangeReport/MI_INDEX?response=json&date=20210707&type=ALLBUT0999&_=1625637146126'

headers = {'User-Agent': 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_10_1) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/39.0.2171.95 Safari/537.36'}

res = requests.get(url,headers=headers)

import io
import pandas as pd

# 篩選出上市股票的資料
lines = [line for line in res.text.split('\n') if len(line.split(',"'))>=10]

df = pd.read_csv(io.StringIO('\n'.join(lines).replace('=','')))
df = df.rename(columns={'證券代號':'stock_id'}).set_index('stock_id')
df = df.applymap(lambda s:pd.to_numeric(str(s).replace(',',''),errors='coerce')).dropna(axis=1,how='all')
df

import requests

    'encodeURIComponent': '1',
    'step': '1',

import requests

url = 'https://www.taifex.com.tw/cht/3/futContractsDate'
headers = {'User-Agent': 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_10_1) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/39.0.2171.95 Safari/537.36'}

payload = {
    'queryDate': '2021/07/05'
}

res = requests.post(url,headers=headers,data=payload)

df = pd.read_html(res.text)[3][list(range(1,15))]

col = df.iloc[:3].fillna('').sum()
df.columns = col
df = df.iloc[3:-4].set_index(['商品名稱','身份別'])
df = df.applymap(lambda s:pd.to_numeric(s,errors='coerce'))

# 分析外資交易
df.loc[df.index.get_level_values('身份別')=='外資']

期貨選擇權下載檔爬蟲

  • 檢查網頁型態 ➜ 表格 ➜ pd.read

import requests
import datetime

url = 'https://www.taifex.com.tw/cht/3/futAndOptDateDown'
headers = {'User-Agent': 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_10_1) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/39.0.2171.95 Safari/537.36'}

end_date = datetime.date(2010,6,23)
start_date = end_date-datetime.timedelta(360*3)
# Form Data
payload = {
    'firstDate': start_date.strftime('%Y/%m/%d')+' 00:00',
    'lastDate': end_date.strftime('%Y/%m/%d')+' 00:00',
    'queryStartDate': start_date.strftime('%Y/%m/%d'),
    'queryEndDate': end_date.strftime('%Y/%m/%d'),
}

res = requests.post(url,headers=headers,data=payload)

import pandas as pd
import io 

df = pd.read_csv(io.StringIO('/n'.join([line for line in res.text.split('\n')]).replace('/n',''))).set_index('日期')
df[df.columns[1:]] = df[df.columns[1:]].apply(lambda s:pd.to_numeric(s,errors='coerce'))

df

Pchome爬蟲

  • 檢查網頁型態 ➜ 非表格 ➜ res.json / beautifulsoup

import requests

url = 'https://ecshweb.pchome.com.tw/search/v3.3/all/results?q=Nike%E7%90%83%E9%9E%8B&page=1&sort=sale/dc'

headers = {'User-Agent': 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_10_1) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/39.0.2171.95 Safari/537.36'}

res = requests.get(url,headers=headers)

products = {}
for prod in res.json()['prods']:
    products[prod['name']] = prod['price']
    
df = pd.DataFrame(products,index=['price']).transpose()
df
